#### LeetCode代码

```cpp
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int n = nums.size();
        int sum = 0,left =0,ans = n+1;
        //i是向右移动的右端点
        for(int i= 0;i<n;i++){
            sum += nums[i];
            while(sum-nums[left]>=target){
                sum -=nums[left];
                left++;
            }
            if(sum>=target){
                ans = min(ans,i-left+1);
            }
        }
        return ans<=n?ans:0;
    }
};```
