class Solution {
public:
    int numSubarrayProductLessThanK(vector<int>& nums, int k) {
        int n = nums.size();
        int ans = 0;
        int left = 0;
        if(k<=1){
            return 0;
        }
        int x = 1;
        for(int right = 0;right<n;right++){
            x *= nums[right];
            while(x>=k){
                x/=nums[left];
                left++;
            }
            ans += right -left +1;
        }
        return ans;
    }
};
