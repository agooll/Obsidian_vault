### 题目
```TXT
给你一个字符串数组，请你将 字母异位词 组合在一起。可以按任意顺序返回结果列表。

**示例 1:**

**输入:** strs = ["eat", "tea", "tan", "ate", "nat", "bat"]

**输出:** [["bat"],["nat","tan"],["ate","eat","tea"]]

**解释：**

- 在 strs 中没有字符串可以通过重新排列来形成 `"bat"`。
- 字符串 `"nat"` 和 `"tan"` 是字母异位词，因为它们可以重新排列以形成彼此。
- 字符串 `"ate"` ，`"eat"` 和 `"tea"` 是字母异位词，因为它们可以重新排列以形成彼此。

**示例 2:**

**输入:** strs = [""]

**输出:** [[""]]

**示例 3:**

**输入:** strs = ["a"]

**输出:** [["a"]]

**提示：**

- `1 <= strs.length <= 104`
- `0 <= strs[i].length <= 100`
- `strs[i]` 仅包含小写字母
```
### ACM解法
```C++
#include <iostream>
#include <vector>
#include <string>
#include <unordered_map>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector<string> strs(n);

    for (int i = 0; i < n; i++) {
        cin >> strs[i];
    }

    unordered_map<string, vector<string>> mp;

    for (int i = 0; i < n; i++) {
        string s = strs[i];

        string key = s;
        sort(key.begin(), key.end());

        mp[key].push_back(s);
    }

    for (auto& p : mp) {
        for (int i = 0; i < p.second.size(); i++) {
            cout << p.second[i];

            if (i + 1 < p.second.size()) {
                cout << " ";
            }
        }
        cout << endl;
    }

    return 0;
}
```

### ACM解法（逐行注释版）
```C++
#include <iostream>                 // 引入输入输出流库，后面要用 cin 和 cout
#include <vector>                   // 引入 vector 容器
#include <string>                   // 引入 string 字符串类型
#include <unordered_map>            // 引入哈希表 unordered_map
#include <algorithm>                // 引入 sort 排序函数
using namespace std;                // 使用标准命名空间，后面就不用反复写 std::

int main() {                        // 主函数，程序从这里开始执行
    int n;                          // 定义一个整数 n，表示有多少个字符串
    cin >> n;                       // 从输入中读入 n

    vector<string> strs(n);         // 创建一个长度为 n 的字符串数组 strs

    for (int i = 0; i < n; i++) {   // 循环 n 次，依次读入每个字符串
        cin >> strs[i];             // 把第 i 个输入字符串存到 strs[i] 中
    }

    unordered_map<string, vector<string>> mp; // 定义哈希表：key 是排序后的字符串，value 是同组异位词列表

    for (int i = 0; i < n; i++) {   // 再遍历一遍所有字符串，开始分组
        string s = strs[i];         // 取出当前字符串，保存到 s 中

        string key = s;             // 复制一份当前字符串到 key，后面要对 key 排序
        sort(key.begin(), key.end()); // 对 key 排序，让异位词得到相同的“特征串”

        mp[key].push_back(s);       // 把原字符串 s 放进 key 对应的那一组里
    }

    for (auto& p : mp) {            // 遍历哈希表中的每一组，p 表示一个键值对
        for (int i = 0; i < p.second.size(); i++) { // 遍历这一组中的每个字符串
            cout << p.second[i];    // 输出当前这一组中的第 i 个字符串

            if (i + 1 < p.second.size()) { // 如果后面还有字符串
                cout << " ";        // 就输出一个空格，方便分隔
            }
        }
        cout << endl;               // 当前这一组输出完成后换行
    }

    return 0;                       // 程序正常结束，返回 0
}
```
### LeetCode写法
```C++
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        unordered_map<string, vector<string>> mp;

        for (string s : strs) {
            string key = s;
            sort(key.begin(), key.end());

            mp[key].push_back(s);
        }

        vector<vector<string>> ans;

        for (auto& pair : mp) {
            ans.push_back(pair.second);
        }

        return ans;
    }
};
```
